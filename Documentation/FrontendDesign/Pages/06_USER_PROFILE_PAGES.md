## 9. State Management

### 9.1 User Profile State

```typescript
// Current user query
const { data: currentUser } = useQuery({
  queryKey: ['users', 'me'],
  queryFn: () => api.users.me(),
});

// Public profile query
const { data: profile } = useQuery({
  queryKey: ['users', username],
  queryFn: () => api.users.getByUsername(username),
});

// Follow mutation
const followMutation = useMutation({
  mutationFn: (userId: string) => api.users.follow(userId),
  onSuccess: () => {
    queryClient.invalidateQueries(['users', username]);
  },
});
```

### 9.2 Settings State

```typescript
// Settings form with auto-save
function useSettingsForm<T>(section: string, schema: ZodSchema<T>) {
  const { data: settings } = useQuery({
    queryKey: ['settings', section],
    queryFn: () => api.settings.get(section),
  });
  
  const form = useForm<T>({
    resolver: zodResolver(schema),
    defaultValues: settings,
  });
  
  const mutation = useMutation({
    mutationFn: (data: T) => api.settings.update(section, data),
    onSuccess: () => {
      toast.success('Settings saved');
      queryClient.invalidateQueries(['settings', section]);
    },
  });
  
  return { form, mutation };
}
```
