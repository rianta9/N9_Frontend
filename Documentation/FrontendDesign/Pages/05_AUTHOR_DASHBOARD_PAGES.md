### 8.5 Auto-save

```typescript
// Auto-save draft every 30 seconds
function useAutoSave(content: string, chapterId?: string) {
  const debouncedContent = useDebounce(content, 30000);
  const { mutate: saveDraft } = useSaveDraftMutation();
  
  useEffect(() => {
    if (debouncedContent) {
      saveDraft({
        chapterId,
        content: debouncedContent,
        savedAt: new Date().toISOString(),
      });
    }
  }, [debouncedContent, chapterId, saveDraft]);
  
  // Also save on page leave
  useEffect(() => {
    const handleBeforeUnload = (e: BeforeUnloadEvent) => {
      if (hasUnsavedChanges) {
        e.preventDefault();
        e.returnValue = '';
      }
    };
    window.addEventListener('beforeunload', handleBeforeUnload);
    return () => window.removeEventListener('beforeunload', handleBeforeUnload);
  }, [hasUnsavedChanges]);
}
```
